useMemo^

const value = computeExpensiveValue(a, b);





const memoizedValue = useMemo(
  () => computeExpensiveValue(a, b),
  [a, b]
);





const memoizedObject = useMemo(
  () => ({ firstName: 'foo', lastName: 'bar' }),
  []
);

useCallback^




const callback = () => {
  doSomething(a, b);
};




const memoizedCallback = useCallback(() => {
  doSomething(a, b);
}, [a, b]);
